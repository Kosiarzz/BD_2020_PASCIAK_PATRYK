create or replace PROCEDURE DODAJ_LOT
(numer_lotuu IN KLIENT.IMIE%TYPE,
ID_samolotuu IN KLIENT.NAZWISKO%TYPE,
miejsca IN KLIENT.ADRES%TYPE,
startt IN KLIENT.PESEL%TYPE,
ladowanie IN KLIENT.TELEFON%TYPE,
powrot IN KLIENT.LOGIN%TYPE,
statuss IN KLIENT.TELEFON%TYPE,
zz IN KLIENT.TELEFON%TYPE,
do IN KLIENT.TELEFON%TYPE)
AS 
BEGIN
  INSERT INTO LOTY (NUMER_LOTU,ID_SAMOLOT,DOSTEPNE_MIEJSCA,STARTOWANIE,LADOWANIE,POWROT,STATUS,Z,DO)
  VALUES (numer_lotuu,ID_samolotuu,miejsca,startt,ladowanie,powrot,statuss,zz,do);
  COMMIT;
END DODAJ_LOT;
