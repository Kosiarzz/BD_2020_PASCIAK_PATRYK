create or replace PROCEDURE INSERT_REJESTRACJA_DANE
(imiee IN KLIENT.IMIE%TYPE,
nazwiskoo IN KLIENT.NAZWISKO%TYPE,
adress IN KLIENT.ADRES%TYPE,
pesell IN KLIENT.PESEL%TYPE,
telefonn IN KLIENT.TELEFON%TYPE,
loginn IN KLIENT.LOGIN%TYPE)
AS 
BEGIN
  INSERT INTO KLIENT (IMIE,NAZWISKO,ADRES,PESEL,TELEFON,LOGIN)
  VALUES (imiee,nazwiskoo,adress,pesell,telefonn,loginn);
  COMMIT;
END INSERT_REJESTRACJA_DANE;
